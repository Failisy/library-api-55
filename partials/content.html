const apiBase = "https://army-library55.wofyf789.workers.dev/"; document.addEventListener("DOMContentLoaded", async () => { try { await loadContent(); fetchBooks(); } catch (error) { console.error("페이지 로드 중 오류 발생:", error); } }); async function loadContent() { const header = document.getElementById("header"); const content = document.getElementById("content"); if (header && content) { header.innerHTML = await (await fetch("/library-api-55/partials/header.html")).text(); content.innerHTML = await (await fetch("/library-api-55/partials/content.html")).text(); } } async function fetchBooks() { try { await new Promise(resolve => setTimeout(resolve, 500)); // Ensure content is loaded const tbody = document.getElementById("book-list"); if (!tbody) { console.error("도서 목록을 표시할 요소를 찾을 수 없습니다."); return; } const response = await fetch(apiBase + "books"); if (!response.ok) { throw new Error("서버 응답 오류"); } const books = await response.json(); tbody.innerHTML = ""; books.forEach(book => { const row = ` ${book.ID} ${book.제목} ${book.저자} ${book.출판사} ${book.ISBN} ${book.대출_가능_여부} `; tbody.innerHTML += row; }); } catch (error) { console.error("도서 목록을 불러오는 중 오류 발생:", error); } }
